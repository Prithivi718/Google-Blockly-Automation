A_raw = None
A_parsed = None
A_temp = None
A_i = None
A_char = None
A = None
largest_side = None
side = None

def text_prompt(msg):
  try:
    return raw_input(msg)
  except NameError:
    return input(msg)

def upRange(start, stop, step):
  while start <= stop:
    yield start
    start += abs(step)

def downRange(start, stop, step):
  while start >= stop:
    yield start
    start -= abs(step)


A_raw = text_prompt('')
A_parsed = []
A_temp = ''
A_i_end = len(A_raw)
for A_i in (1 <= A_i_end) and upRange(1, A_i_end, 1) or downRange(1, A_i_end, 1):
  A_char = A_raw[int(A_i - 1)]
  if A_char == ' ':
    if A_temp != '':
      A_parsed.append(int(A_temp))
      A_temp = ''
if A_temp != '':
  A_parsed.append(int(A_temp))
A = A_parsed
largest_side = A[int((0 + 1) - 1)]
for side in A:
  if side > largest_side:
    largest_side = side
print(largest_side)
