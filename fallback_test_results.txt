def move_zerosToEnd(arr):
    non_zero = [x for x in arr if x != 0]
    zeros = [x for x in arr if x == 0]
    result = non_zero + zeros
    return result

arr = [1, 2, 0, 4, 3, 0, 5, 0]
output = move_zerosToEnd(arr)
print(output)

import random

numbers = [random.randint(1, 100) for _ in range(10)]
largest = numbers[0]
for num in numbers:
    if num > largest:
        largest = num
print(f'Largest number is {largest}')
